import streamlit as st
import os
import json

st.set_page_config(page_title="Stellar Atlas", layout="wide")
st.title("🌟 Stellar Evolution Atlas")

# ✅ YOU must place folders like data/sun_star, data/10M_solar, etc.
star_dirs = sorted(os.listdir("work"))

# ✅ The folder names appear in this dropdown
selected_star = st.selectbox("Choose a Star", star_dirs)

# ✅ YOU must provide metadata.json inside each folder
with open(f"data/{selected_star}/metadata.json") as f:
    meta = json.load(f)

# ✅ YOU must write these in metadata.json
st.subheader(f"{meta['name']} — {meta['mass']} M☉")
st.markdown(meta["description"])

# ✅ YOU must provide hr.png, temp_rho.png
st.image(f"data/{selected_star}/hr.png", caption="HR Diagram")
